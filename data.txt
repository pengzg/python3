import api from '../api/api';
import wepy from 'wepy';
import zanTab from '../components/common/zan-tab';

import tip from '../utils/tip';
import bottomLoadMore from '../components/common/bottomLoadMore';
import placeholder from '../components/common/placeholder';
export default class WaterTicket extends wepy.page {
  config = {
    navigationBarTitleText: '水票',
    enablePullDownRefresh: true
  }
  data = function(){
    return {
        bottoms:180,
        startX: 0,
        choose: false,
        nodata:false,
        tab1: {
          list: [
            {
              id: 0,
              title: '购买水票'
            },
            {
              id: 1,
              title: '我的水票'
            }
          ],
          activeIdx: 1,
          scroll: false
        },
        tab2: {
          list: [
           
     
            {
              useState: 2,
              title: '使用中'
            },
            {
              useState: 3,
              title: '使用完'
            },
            {
              useState: 4,
              title: '已过期' 
            }
          ],
          activeIdx: 1,
          scroll: false
        },
        //购买水票
        dataListGMSP:[],
        dataList: [],
        pageState: 0,
        userInfo:{},
        pager : {page : 1,rows:10,orderStatus:0},
        pager2 : {page : 1,rows:10},
        shareItem:null,
        buyItem:null,
        buyNum:1,
        totalAmount:0,
        accountBalance:0,
        params:{},
        isShowNew:0
        
        
    }
  }
  components = {
    zanTab1: zanTab,
    zanTab2: zanTab,
    scrolltop:scrolltop,
    bottomLoadMore: bottomLoadMore,
    placeholder: placeholder
  }
  events = {
    async zanTabChange(activeIdx, event) {
      console.log(activeIdx,1);
      tip.loading();
      this.pageState = 0;
      this.pager.page =1;
      let { componentId } = event.source
      this[componentId].activeIdx = activeIdx;
      if(componentId=='tab1'){
        
        this.tab2.activeIdx = 1;
        if(activeIdx==1){
          
          // 我的水票 
          this.pager.tm_use_state = this.tab2.list[1].useState;
          let result = await this.queryTicketMainList(this.pager);
          this.dataList = result.data.data.rows;
          if (this.pager.page ==1) {
            this.nodata = this.dataList.length==0;
          }

        }else{
          // 购买水票
          this.dataListGMSP = await this.getActivityList(this.pager2);
          this.nodata = this.dataListGMSP.length ==0;
          if (!this.nodata) {
            this.buyItem = this.dataListGMSP[0];
            this.totalAmount = this.buyItem.total_amount*this.buyNum;
          }
        }
      }

      // 我的水票里面的
      if(componentId=='tab2'){
        this.dataList = [];
        this.pager.tm_use_state = this.tab2.list[activeIdx].useState;
        let result = await this.queryTicketMainList(this.pager);
        this.dataList = result.data.data.rows;
        if (this.pager.page ==1) {
          this.nodata = this.dataList.length==0;
        }

      }

      tip.loaded();
      this.$apply()
    }

      wx.navigateTo({

      /* console.log(item.od_mainid);
       wx.navigateTo({
        url: 
      }) */
      if (item.tm_total_num>item.tm_available_num) {
        let url = "/pages/ticket/ticketValidation?ticketId="+item.tm_id;
        wx.navigateTo({
          url: url
        })
      }
      
    },
    goGoodsDetail(item){
     // let url = '/pages/goodsDetail?pmInfo='+JSON.stringify(item);

      console.log(url);
        wx.navigateTo({
          url: url
        })
    },
    selectShareItem(obj) {

        return false;
      }
      this.buyNum = 1;
      this.buyItem = obj;
      this.totalAmount = this.buyItem.total_amount*this.buyNum;
    },

      if (this.buyNum==1) {
        return false;
      }
      this.buyNum--;
      this.totalAmount = this.buyItem.total_amount*this.buyNum;
    },
    plusBuy(obj) {
      this.buyNum++;
      this.totalAmount = this.buyItem.total_amount*this.buyNum;
    },
   
  }
 async onShow(){

  }

      if(this.pageState == 1 || this.pageState == 2){
        return;
      }
      tip.loading();
      this.pageState = 1;
      this.pager.page ++;
      
      let result = await this.queryTicketMainList(this.pager);

      if(tempList.length==0){
        this.pageState = 2;
        this.data.pager.page --;
        this.$apply();
      } else {
        this.pageState = 0;
        this.dataList = this.dataList.concat(tempList);
      }
      tip.loaded();
      
      this.$apply();
      // 
    } else {
      
       if(this.pageState == 1 || this.pageState == 2){
        return;
      }
       tip.loading();  
      this.pageState = 1;
      this.pager2.page ++;
      
      let result = await this.getActivityList(this.pager2);

      if(tempList.length==0){
        this.pageState = 2;
        this.data.pager2.page --;
        this.$apply();
      } else {
        this.pageState = 0;
        this.dataListGMSP = this.dataListGMSP.concat(tempList);
      }
      tip.loaded();
      this.$apply();
      
      
    }
  }

 //   this.accountBalance = await this.queryAccountBlance();
    if(this.tab1.activeIdx == 0){
        this.pager2.page=1;
        this.dataListGMSP = await this.getActivityList(this.pager2);
      }else{
        this.pager.page = 1;

        if(result.data.code!=200){
          return;
        }
        this.dataList = result.data.data.rows;
        this.nodata = this.dataList.length==0;
      }
      this.pageState = 0;
      this.$apply();
  }
  async onLoad(option){
    try{
     // this.accountBalance = await this.queryAccountBlance();
      this.userInfo = wx.getStorageSync('userInfo');
      this.pager2.areaCode = this.userInfo.locationCode;
      if(this.tab1.activeIdx == 0){
        this.dataListGMSP = await this.getActivityList(this.pager2);        
      }else{
        if(option.tm_use_state){
          this.pager.tm_use_state = option.useState;
        }else{
          this.pager.tm_use_state = 2;
        }
        this.tab2.activeIdx = this.pager.tm_use_state-2;   
        let result = await this.queryTicketMainList(this.pager);
        if(result.data.code!=200){
          return;
        }
        this.dataList = result.data.data.rows;
        this.nodata = this.dataList.length==0;
        this.$apply();
      }
    }catch(e){
        console.error(e)
    }
  }
}
